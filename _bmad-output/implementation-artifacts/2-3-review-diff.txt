+ 255 export async function validateCode(rawCode: string): Promise<ValidationCodeResult> {
+ 256   const code = normalizeCode(rawCode)
+ 257   if (!code) {
+ 258     return { valid: false, reason: 'invalid_format' }
+ 259   }
+ 260   if (code.length !== CODE_LENGTH) {
+ 261     return { valid: false, reason: 'invalid_format' }
+ 262   }
+ 263   if (!CODE_REGEX.test(code)) {
+ 264     return { valid: false, reason: 'invalid_chars' }
+ 265   }
+ 266 
+ 267   // RLS bloque la lecture anon de parrainages_codes (auth.uid() = user_id requis).
+ 268   // La fonction étant appelée depuis le formulaire d'inscription non authentifié,
+ 269   // on utilise le service role pour cette lecture publique de validation.
+ 270   const supabaseAdmin = await createClient({ serviceRole: true })
+ 271 
+ 272   const { data: row } = await supabaseAdmin
+ 273     .from('parrainages_codes')
+ 274     .select('user_id')
+ 275     .eq('code', code)
+ 276     .maybeSingle()
+ 277 
+ 278   if (!row?.user_id) {
+ 279     return { valid: false, reason: 'unknown_code' }
+ 280   }
+ 281 
+ 282   const { data: marraine } = await supabaseAdmin
+ 283     .from('users')
+ 284     .select('first_name, accompagnantes_profiles!inner(validation_status)')
+ 285     .eq('id', row.user_id)
+ 286     .maybeSingle()
+ 287 
+ 288   const profileRow = marraine?.accompagnantes_profiles as
+ 289     | { validation_status?: string }
+ 290     | { validation_status?: string }[]
+ 291     | undefined
+ 292   const validationStatus = Array.isArray(profileRow)
+ 293     ? profileRow[0]?.validation_status
+ 294     : profileRow?.validation_status
+ 295 
+ 296   if (validationStatus !== 'valide') {
+ 297     return { valid: false, reason: 'marraine_not_validated' }
+ 298   }
+ 299 
+ 300   return {
+ 301     valid: true,
+ 302     marraineId: row.user_id,
+ 303     marraineFirstName: (marraine?.first_name as string) || '',
+ 304   }
+ 305 }

+ 499 export async function confirmParrainageOnSuccess(
+ 500   sessionId: string
+ 501 ): Promise<ConfirmParrainageOnSuccessResult> {

+ 503 
+ 504   const supabase = await createClient()
+ 505   const {
+ 506     data: { user },
+ 507   } = await supabase.auth.getUser()
+ 508   if (!user) return { ok: false, reason: 'not_authenticated' }
+ 509 
+ 510   let session
+ 511   try {
+ 512     session = await stripe.checkout.sessions.retrieve(sessionId)
+ 513   } catch {
+ 514     return { ok: false, reason: 'stripe_session_not_found' }
+ 515   }
+ 516 
+ 517   const parrainageCode = (session.metadata?.parrainage_code || '').toString()
+ 518   if (!parrainageCode) {
+ 519     return { ok: false, reason: 'no_parrainage_metadata' }
+ 520   }
+ 521 
+ 522   const isComplete = session.status === 'complete' || session.payment_status === 'paid'
+ 523   if (!isComplete) {
+ 524     return { ok: false, reason: 'session_not_complete' }
+ 525   }
+ 526   // Sécurité : la metadata user_id doit être présente ET correspondre à l'utilisateur courant.
+ 527   const sessionUserId = (session.metadata?.user_id || '').toString()
+ 528   if (sessionUserId !== user.id) {
+ 529     return { ok: false, reason: 'user_mismatch' }
+ 530   }
+ 531 
+ 532   const code = normalizeCode(parrainageCode)
+ 533   const supabaseAdmin = await createClient({ serviceRole: true })
+ 534 
+ 535   const { data: parrainage } = await supabaseAdmin
+ 536     .from('parrainages')
+ 537     .select('id, statut, marraine_id, filleule_id')
+ 538     .eq('code', code)
+ 539     .eq('filleule_id', user.id)
+ 540     .order('filleule_inscrite_at', { ascending: false })
+ 541     .limit(1)
+ 542     .maybeSingle()
+ 543 
+ 544   if (!parrainage) {
+ 545     return { ok: false, reason: 'parrainage_not_found' }
+ 546   }
+ 547 
+ 548   if (parrainage.statut !== 'inscrite') {
+ 549     return { ok: true, alreadyDone: true }
+ 550   }
+ 551 
+ 552   // Vérification que la marraine est toujours validée au moment du paiement.
+ 553   // Si elle a été déchue entre l'inscription de la filleule et le paiement, on
+ 554   // refuse la validation automatique par parrainage.
+ 555   const { data: marraineProfile } = await supabaseAdmin
+ 556     .from('accompagnantes_profiles')
+ 557     .select('validation_status')
+ 558     .eq('user_id', parrainage.marraine_id)
+ 559     .maybeSingle()
+ 560   if (marraineProfile?.validation_status !== 'valide') {
+ 561     return { ok: false, reason: 'marraine_no_longer_validated' }
+ 562   }
+ 563 
+ 564   // Décision 3a : refuser la validation et le log si le profil filleule est absent.
+ 565   const { data: filleuleProfile } = await supabaseAdmin
+ 566     .from('accompagnantes_profiles')
+ 567     .select('id')
+ 568     .eq('user_id', user.id)
+ 569     .single()
+ 570 
+ 571   if (!filleuleProfile?.id) {
+ 572     return { ok: false, reason: 'profile_not_found' }
+ 573   }
+ 574 
+ 575   const nowIso = new Date().toISOString()
+ 576 
+ 577   // Idempotence transactionnelle (optimistic locking) : on tente le passage à 'abonnee'
+ 578   // EN PREMIER avec un filtre statut='inscrite'. Si 0 ligne affectée, c'est qu'un autre
+ 579   // appel concurrent a déjà fait la transition — on abandonne pour éviter doubles emails/logs.
+ 580   const { data: lockedRows } = await supabaseAdmin
+ 581     .from('parrainages')
+ 582     .update({
+ 583       statut: 'abonnee',

+async function captureParrainageFingerprint(
+  supabase: ReturnType<typeof getSupabaseAdmin>,
+  subscription: Stripe.Subscription,
+): Promise<void> {
+  try {
+    const rawCode = subscription.metadata?.parrainage_code
+    const userId = subscription.metadata?.user_id
+    if (!rawCode || !userId) return
+
+    const parrainageCode = normalizeParrainageCode(rawCode)
+    if (!parrainageCode) return
+
+    const { data: existingParrainage } = await supabase
+      .from('parrainages')
+      .select('id, stripe_fingerprint')
+      .eq('code', parrainageCode)
+      .eq('filleule_id', userId)
+      .eq('statut', 'inscrite')
+      .maybeSingle()
+
+    if (!existingParrainage || existingParrainage.stripe_fingerprint != null) return
+
+    const defaultPm = subscription.default_payment_method
+    const pmId = typeof defaultPm === 'string' ? defaultPm : defaultPm?.id ?? null
+    if (!pmId) return
+
+    const pm = await stripe.paymentMethods.retrieve(pmId)
+    const fingerprint = pm.card?.fingerprint ?? null
+    if (!fingerprint) return
+
+    await supabase
+      .from('parrainages')
+      .update({ stripe_fingerprint: fingerprint })
+      .eq('code', parrainageCode)
+      .eq('filleule_id', userId)
+      .eq('statut', 'inscrite')
+      .is('stripe_fingerprint', null)
+
+    // Story 2.3 AC3 : déclencher le second passage de détection si le parrainage
+    // est encore inscrit et sans blocage.
+    const { data: refreshedParrainage } = await supabase
+      .from('parrainages')
+      .select('id, marraine_id, filleule_id, stripe_fingerprint, flag_suspicion, statut, blocage_raison')
+      .eq('id', existingParrainage.id)
+      .maybeSingle()
+    if (
+      refreshedParrainage &&
+      refreshedParrainage.statut === 'inscrite' &&
+      !refreshedParrainage.blocage_raison
+    ) {
+      await detectBlacklistAtWebhook(supabase, {
+        id: refreshedParrainage.id,
+        marraine_id: refreshedParrainage.marraine_id,
+        filleule_id: refreshedParrainage.filleule_id,
+        stripe_fingerprint: refreshedParrainage.stripe_fingerprint,
+        flag_suspicion: refreshedParrainage.flag_suspicion,
+      })
+    }
+  } catch (err) {
+    // best-effort, ne doit jamais bloquer le flow webhook
+    console.error('[parrainage_fingerprint][subscription_updated]', err)
+  }
